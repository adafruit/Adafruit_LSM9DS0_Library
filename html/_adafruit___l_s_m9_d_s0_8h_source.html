222  Sensor(Adafruit_LSM9DS0 *parent, lsm9ds0_read_func readFunc,
223  lsm9ds0_get_event_func eventFunc, lsm9ds0_get_sensor_func sensorFunc)
224  : _parent(parent), _readFunc(readFunc), _eventFunc(eventFunc),
225  _sensorFunc(sensorFunc) {}
229  virtual bool getEvent(sensors_event_t *event) {
230  /* Take new reading. */
231  (_parent->*_readFunc)();
232  /* Fill in event data. */
233  (_parent->*_eventFunc)(event, millis());
234  return true;
235  }
236 
239  virtual void getSensor(sensor_t *sensor) {
240  /* Fill in sensor metadata. */
241  (_parent->*_sensorFunc)(sensor);
242  }
